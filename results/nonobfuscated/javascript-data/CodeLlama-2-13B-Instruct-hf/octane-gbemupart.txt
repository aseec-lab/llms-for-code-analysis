function base64_decode(str) {
  var result = "";
  var table = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
  for (var i = 0; i < str.length;) {
    var sextet_a = table.indexOf(str.charAt(i++));
    var sextet_b = table.indexOf(str.charAt(i++));
    var sextet_c = table.indexOf(str.charAt(i++));
    var sextet_d = table.indexOf(str.charAt(i++));
    var triple = (sextet_a << 18) + (sextet_b << 12) +
                 (sextet_c << 6) + sextet_d;
    if (sextet_a == -1) {
      continue;
    }
    if (sextet_c == -1) {
      result += String.fromCharCode((triple >> 16) & 255);
      break;
    }
    result += String.fromCharCode((triple >> 16) & 255,
                                  (triple >> 8) & 255,
                                  triple & 255);
  }
  return result;
}
